## Template 1: PostToolUse Validator

```python
#!/usr/bin/env python3
"""
Auto-generated by /av improve /{SKILL_NAME}
Validates {SKILL_NAME} tool output for errors
"""
import json
import sys
from pathlib import Path

def validate_output(data: dict) -> dict:
    """Validate tool output based on skill requirements."""
    tool_name = data.get("tool_name", "")
    tool_output = data.get("tool_output", "")

    # CUSTOMIZE: Add skill-specific validation
    if tool_name == "Write":
        # Check file was created
        filepath = data.get("tool_input", {}).get("path", "")
        if not Path(filepath).exists():
            return {
                "hookSpecificOutput": {
                    "additionalContext": "❌ File write failed. Verify path and retry."
                }
            }

    if tool_name == "Bash":
        # Check for error keywords
        if any(err in tool_output.lower() for err in ["error", "failed", "exception"]):
            return {
                "hookSpecificOutput": {
                    "additionalContext": f"❌ Command failed: {tool_output[:100]}"
                }
            }

    # Success - no intervention needed
    return {}

def main():
    input_data = json.loads(sys.stdin.read())
    result = validate_output(input_data)
    print(json.dumps(result))
    sys.exit(0)

if __name__ == "__main__":
    main()
```

## Template 2: State Manager

```python
#!/usr/bin/env python3
"""
State manager for {SKILL_NAME} workflow
Auto-generated by /av improve /{SKILL_NAME}
"""
from pathlib import Path
import hashlib
import json
import time

# Instance isolation - different CWD = different state file
instance_id = hashlib.md5(str(Path.cwd()).encode()).hexdigest()[:8]
STATE_DIR = Path("P://.claude/hooks/state")
STATE_FILE = STATE_DIR / f"{SKILL_NAME}_{instance_id}.json"

# Note: State stored centrally in .claude/hooks/state/ (skill-internal hooks reference it)

def write_state(phase: str, allowed_tools: list, expires_in: int = 300):
    """Write workflow state with expiration."""
    state = {
        "phase": phase,
        "allowed_tools": allowed_tools,
        "expires_at": time.time() + expires_in,
        "skill": "{SKILL_NAME}",
        "instance": instance_id
    }
    STATE_DIR.mkdir(parents=True, exist_ok=True)
    with open(STATE_FILE, "w") as f:
        json.dump(state, f, indent=2)

def read_state() -> dict | None:
    """Read state if exists and not expired."""
    if not STATE_FILE.exists():
        return None

    with open(STATE_FILE) as f:
        state = json.load(f)

    # Check expiration
    if time.time() > state.get("expires_at", 0):
        STATE_FILE.unlink()
        return None

    return state

def clear_state():
    """Remove state file."""
    if STATE_FILE.exists():
        STATE_FILE.unlink()
```

## Template 3: PreToolUse Gate

```python
#!/usr/bin/env python3
"""
PreToolUse gate for {SKILL_NAME}
Blocks tools not allowed in current phase
Auto-generated by /av improve /{SKILL_NAME}
"""
import json
import sys
from pathlib import Path
sys.path.insert(0, str(Path(__file__).parent))
from state_manager_{SKILL_NAME} import read_state

def main():
    input_data = json.loads(sys.stdin.read())
    tool_name = input_data.get("tool_name", "")

    state = read_state()
    if not state:
        # No state = allow all (or customize this)
        sys.exit(0)

    allowed = state.get("allowed_tools", [])
    phase = state.get("phase", "unknown")

    if tool_name not in allowed:
        print(json.dumps({
            "hookSpecificOutput": {
                "additionalContext": f"⛔ Tool '{tool_name}' not allowed in phase '{phase}'. Allowed: {allowed}"
            }
        }))
        sys.exit(2)  # Block

    sys.exit(0)  # Allow

if __name__ == "__main__":
    main()
```

## Template 4: PostToolUse Transition

```python
#!/usr/bin/env python3
"""
PostToolUse transition for {SKILL_NAME}
Advances workflow phase after successful steps
Auto-generated by /av improve /{SKILL_NAME}
"""
import json
import sys
from pathlib import Path
sys.path.insert(0, str(Path(__file__).parent))
from state_manager_{SKILL_NAME} import write_state, read_state

def main():
    input_data = json.loads(sys.stdin.read())
    tool_name = input_data.get("tool_name", "")

    state = read_state()
    if not state:
        sys.exit(0)

    phase = state.get("phase", "")

    # CUSTOMIZE: Define phase transitions
    if phase == "init" and tool_name == "Skill":
        # After skill loads, allow execution tools
        write_state(phase="execute", allowed_tools=["Bash", "Task", "Write"])

    elif phase == "execute" and tool_name in ["Bash", "Task"]:
        # After execution, allow validation tools
        write_state(phase="validate", allowed_tools=["Read", "View"])

    sys.exit(0)

if __name__ == "__main__":
    main()
```

## Template 5: Execution Skill SKILL.md

Use this template for EXECUTION type skills that invoke external tools/CLIs:

```markdown
---
name: {SKILL_NAME}
version: 1.0.0
description: {DESCRIPTION}
category: command
triggers:
  - /{SKILL_NAME}
aliases:
  - /{SKILL_NAME}
  - /{ALIAS}
suggest:
  - {RELATED_SKILL}
---

# /{SKILL_NAME}

## ⚡ EXECUTION DIRECTIVE

**When invoked, IMMEDIATELY execute:**

\`\`\`bash
python "P:///.claude//{PATH}\\{SCRIPT}.py" "{{user_query}}" {{options}}
\`\`\`

**MANDATORY:**
1. Parse user's query and any options from the invocation
2. Run the bash command above with actual arguments
3. Wait for command output to complete
4. Report the tool output verbatim

**DO NOT:**
- Provide your own analysis instead of running the command
- Summarize this skill documentation
- Substitute your capabilities for the external tool
- Consider the task complete until bash command output is captured
- Read files and analyze them yourself when user wants external tool output

**DEFAULT (no arguments):**
\`\`\`bash
python "P:///.claude//{PATH}\\{SCRIPT}.py" --help
\`\`\`

**If execution fails:** Report exact error message. Do NOT fabricate results or provide your own analysis as substitute.

## ⚠️ Critical Rules

{SKILL_SPECIFIC_RULES}

## 🔧 Troubleshooting
```

```markdown
**When invoked, follow this workflow:**

### Phase 1: {PHASE_1_NAME}
1. {STEP_1}
2. {STEP_2}
**Success criteria:** {CRITERIA_1}
**Allowed tools:** {TOOLS_1}

### Phase 2: {PHASE_2_NAME}
1. {STEP_3}

**DO NOT:**
- Skip phases or reorder steps
- Proceed to next phase without meeting success criteria
- Use tools not allowed in current phase

**DEFAULT (no arguments):** Start at Phase 1

**If phase fails:** Report which phase, what failed, suggest recovery action.
```
